# 06. Ansible run

## Facts hosts

### facts.yaml
```bash
---
- hosts: "{{ group }}"
  become: True
  tasks:
    - name: Print OS
      debug:
        msg: "{{ ansible_distribution }} : {{ ansible_distribution_version }}"
      tags:
        - os
    - name: Print RAM
      debug:
        msg:
          - "RAM (MB):"
          - "{{ ansible_memory_mb.real }}"
      tags:
        - ram
    - name: Print mounts
      debug:
        msg:
          - "Mount point: {{ item.mount }}"
          - "Size total: {{ (item.size_total / 1024 / 1024 / 1024)|round(2) }} GB"
          - "Size used: {{ ((item.size_total-item.size_available) / 1024 / 1024 / 1024)|round(2) }} GB"
      with_items: "{{ ansible_mounts }}"
      tags:
        - mount
```

### add_or_rm_user.yaml
```bash
---
- hosts: "{{ _hosts }}"
  become: True
  tasks:

  - name: Create user {{ user_add }}
    user:
      name: "{{ user_add }}"
      groups: "{{ group }}"
      state: present
    tags:
      create_user
      full_create_user

  - name: Create ssh dir {{ user_add }}
    file:
      path: /home/{{ user_add }}/.ssh
      state: directory
      owner: "{{ user_add }}"
      group: "{{ user_add }}"
      mode: 0700
    tags:
      mkdir_ssh
      full_create_user

  - name: Copy public key on remoat server
    copy:
      src: /home/krask/.ssh/id_rsa.pub
      dest: /home/{{ user_add }}/.ssh/authorized_keys
      owner: "{{ user_add }}"
      group: "{{ user_add }}"
      mode: 0600
    tags:
      full_create_user

  - name: Add {{ user_add }} to the sudo:nopasswd
    lineinfile:
      path: /etc/sudoers
      regexp: '^{{ user_add }}'
      line: "{{ user_add }} ALL=(ALL) NOPASSWD: ALL"
      state: present
    tags:
      visudo
      full_create_user

  - name: Removing user {{ user_add }}
    user:
      name: "{{ user_add }}"
      state: absent
    tags:
      remove_user

  - name: Testing root access {{ user_add }}
    shell: |
      ls -a /root/
    register: out
    tags:
      testing_root_access

  - name: Print output
    debug:
      msg: "{{ out.stdout_lines }}"
    tags:
      testing_root_access
```
