** Playbook for add the ansible user - add_user.yaml **

```
---
- hosts: all
  vars:
    name_user: ansible_user
  become: yes 

  tasks:  
  - name: create user
    user: 
      name: "{{ name_user }}"
      shell: /bin/bash 
      home: /home/{{name_user}}
  
  - name: user to sudo
    copy:
      dest: "/etc/sudoers.d/{{ name_user }}"
      content: "{{ name_user }} ALL=(ALL)  NOPASSWD: ALL"

  - name: Deploy SSH Key
    authorized_key: 
      user: "{{name_user}}"
      key: "{{ lookup('file', '/home/vagrant/.ssh/id_rsa.pub') }}"
      state: present
```

** Playbook for delete the ansible user - del_user.yaml **

```
---
- hosts: all
  vars:
    name_user: ansible_user
  become: yes
  become_method: sudo  

  tasks:  
  
  - name: delete user
    user:
      name: "{{ name_user }}"
      state: absent
      remove: yes

  - name: delete user in sudo
    file:
      path: /etc/sudoers.d/{{ name_user }}
      state: absent
```
