## Create ansible user

````bash

ansible-playbook play_user.yaml -i inventory/main.yaml -e "user_to_add=ansible_user" -b

PLAY [virtualbox] ***************************************************************************************************************************************************************************************************************************

TASK [Gathering Facts] **********************************************************************************************************************************************************************************************************************
ok: [centos_virt]
ok: [ubuntu_virt]

TASK [Print variables] **********************************************************************************************************************************************************************************************************************
ok: [centos_virt] => {
    "msg": "You requested user ansible_user"
}
ok: [ubuntu_virt] => {
    "msg": "You requested user ansible_user"
}

TASK [Creating user ansible_user] ***********************************************************************************************************************************************************************************************************
ok: [ubuntu_virt]
ok: [centos_virt]

TASK [Set authorized key taken from file] ***************************************************************************************************************************************************************************************************
ok: [ubuntu_virt]
ok: [centos_virt]

TASK [Add user ansible_user to sudo] ********************************************************************************************************************************************************************************************************
ok: [ubuntu_virt]
ok: [centos_virt]

PLAY RECAP **********************************************************************************************************************************************************************************************************************************
centos_virt                : ok=5    changed=0    unreachable=0    failed=0    skipped=0    rescued=0    ignored=0
ubuntu_virt                : ok=5    changed=0    unreachable=0    failed=0    skipped=0    rescued=0    ignored=0


````
