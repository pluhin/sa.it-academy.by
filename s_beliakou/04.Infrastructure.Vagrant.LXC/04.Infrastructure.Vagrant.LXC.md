## LXC


# Config in config.conf

```bash

# Common section
lxc.uts.name = cent
lxc.arch = amd64
lxc.cgroup.memory.limit_in_bytes = 512m
lxc.cgroup.cpuset.cpus = 0
lxc.cgroup.blkio.weight = 500

#
# # Network section
lxc.net.1.type = veth
lxc.net.1.flags = up
lxc.net.1.link = lxcbr0

```
