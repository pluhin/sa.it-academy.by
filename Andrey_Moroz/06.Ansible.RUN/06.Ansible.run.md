       ansible_host: 10.10.10.3

    bastion:
      ansible_host: 172.16.1.2

```yaml

ansible_user: ansible
``` 

### parameters.yaml
```yaml
---
- hosts: "{{group}}"
  tasks:
    - name: OS Version
      debug:
        msg: " {{ ansible_distribution }}-{{ ansible_distribution_version }} "
      tags:
        os::version

    - name: free -h
      shell: |
        free -h
      register: out_free
      tags:
        free::h
    - debug:
         msg: "{{ out_free.stdout_lines }}"
      tags:
        free::h

    - name: df -h
      shell: |
        df -h
      register: out_mount
      tags:
        df::h
    - debug:
        msg: "{{ out_mount.stdout_lines }}"
      tags:
        df::h
```

### ansible_user.yaml
```yaml
---
- hosts: centos_servers
  become: True
  tasks:

  - name: Creating user {{ user_to_add }}
    user:
      name: "{{ user_to_add }}"
      comment: Managed by Ansible
      state: present
    tags:
      create::user

  - name: ADD sudo and create folder /.ssh
    shell: |
      echo "{{ user_to_add }}         ALL=(ALL)       NOPASSWD: ALL" >> /etc/sudoers
    tags:
      add::sudo

  - name: Create dir
    file:
      path: /home/{{ user_to_add }}/.ssh
      state: directory
      owner: "{{ user_to_add }}"
      group: "{{ user_to_add }}"
      mode: '0771'
    tags:
      mkdir

  - name: Removing user {{ user_to_add }}
    user:
      name: "{{ user_to_add }}"
      state: absent
    tags:
      remove::user

  - name: Copy file with owner and permissions
    copy:
      src: /home/andrey_m/authorized_keys
      dest: /home/{{ user_to_add }}/.ssh/authorized_keys
      owner: "{{ user_to_add }}"
      group: "{{ user_to_add }}"
      mode: '0644'
    tags:
      cp::keys

  - name: change chmod
    shell: |
      chmod 711 /home/{{user_to_add}}/.ssh |
      chmod 600 /home/{{user_to_add}}/.ssh/authorized_keys
    tags:
      chmod

  - name: upgrade
    shell: |
      yum -y upgrade
    tags:
      upgrade
```
