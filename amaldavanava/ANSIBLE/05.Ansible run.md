- Playbook for the ansible user

``````yaml
- hosts: ec_htp
  tasks:
    - name: "Creating User {{ user_add }}"
      user:
        name: "{{ user_add }}"
        comment: "Managed by Ansible"
        shell: /bin/bash
        state: present

    - name: Set authorized key taken from file
      authorized_key:
        user: "{{ user_add }}"
        state: present
        key: "{{ lookup('file', 'id_rsa.pub') }}"

    - name: "Set includedir in sudoers"
      lineinfile:
       dest: /etc/sudoers
       line: "#includedir /etc/sudoers.d"
       state: present
       validate: "/usr/sbin/visudo -cf %s"

    - name: "Add user {{ user_add }} to sudo"
      lineinfile:
       path: "/etc/sudoers.d/{{ user_add }}"
       line: '{{ user_add }} ALL=(ALL) NOPASSWD: ALL'
       state: present
       mode: 0440
       create: yes
       validate: 'visudo -cf %s'

```

- Removing user playbook

```yaml
- hosts: ec_htp
  tasks:
    - name: "Removing User {{ user_add }}"
      user:
       name: "{{ user_add }}"
       comment: "Managed by Ansible"
       state: absent

    - name: "Clean {{ user_add }} file in sudoers.d"
      file:
        state: absent
        path: "/etc/sudoers.d/{{ user_add }}" 
  ```
