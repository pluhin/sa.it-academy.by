    - [x] Task steps:

### Edit CasC file

```yaml
jobs:
  - script: >
     job('Job_DSL') {
             scm {
                git {
                  remote {
                     url 'https://github.com/AndreiBaitau/jenkins.git'
               }
            branch('master')
            }
          }
        steps {
          dsl {
            external('Test.groovy')

          }
        }
      }
```
