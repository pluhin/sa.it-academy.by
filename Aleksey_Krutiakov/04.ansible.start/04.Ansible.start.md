- hosts file

```bash

[jump]
jump_host_01 ansible_host=178.124.206.48 ansible_user=jump_sa

[sa_cent]
cent_01 ansible_host=192.168.254.50 ansible_user=root

[sa_ub]
ub_01 ansible_host=192.168.254.51 ansible_user=user

[web:children]
sa_cent
sa_ub

[web:vars]
ansible_ssh_pass=xxxxxx

```
