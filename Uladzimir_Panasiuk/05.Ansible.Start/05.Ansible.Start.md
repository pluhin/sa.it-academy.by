## Remote connection

### Files:

#### Vagrantfile:
```ruby
$deploy_ssh_keys = <<-SCRIPT
cat /tmp/pub_key.pub >> /home/vagrant/.ssh/authorized_keys
chown vagrant:vagrant /home/vagrant/.ssh/authorized_keys
chmod 600 /home/vagrant/.ssh/authorized_keys
SCRIPT

PUB_KEY_PATH = "~/.ssh/id_rsa.pub"

MACHINES = {
  'host1' =>  { vcpu: '1', ram: '512', ip: '10.20.30.20'},
  'host2' =>  { vcpu: '1', ram: '512', ip: '10.20.30.21', box: 'debian/buster64'},
  'jump'  =>  { vcpu: '1', ram: '512', ip: '10.20.30.22', local_ip: '192.168.100.200', local_br_ifname: 'enp3s0'},
}

INSTANCES = MACHINES.flat_map do |server_name, server|
    server.merge({
      :vagrant_name => "#{server_name}",
    })
end

Vagrant.configure("2") do |config|
  config.vm.synced_folder ".", "/vagrant", disabled: true

  INSTANCES.each do |instance|
    config.vm.define instance[:vagrant_name] do |node|
      node.vm.box = "centos/7"
      if instance[:vagrant_name] == "host2"
        node.vm.box = instance[:box]
      end
      node.vm.hostname = instance[:vagrant_name]
      node.vm.network "private_network", ip: instance[:ip]
      if instance[:vagrant_name] == "jump"
        node.vm.network "public_network", ip: instance[:local_ip], bridge: instance[:local_br_ifname]
      end
      node.vm.provider "virtualbox" do |vb|
        vb.gui = false
        vb.memory = instance[:ram]
        vb.cpus = instance[:vcpu]
      end
      node.vm.provision "file", source: PUB_KEY_PATH, destination: "/tmp/pub_key.pub"
      node.vm.provision "shell", inline: $deploy_ssh_keys
    end
  end
end
```

#### ~/.ssh/config
```
Host localvagjump
	User vagrant
	HostName 192.168.100.200
	
Host 10.20.30.20 10.20.30.21
	User vagrant
	ProxyJump localvagjump
```

```ini
debian ansible_host=10.20.30.21
```
