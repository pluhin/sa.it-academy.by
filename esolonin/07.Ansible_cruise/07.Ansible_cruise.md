 # 07.Ansible_cruise

***

* nginx_config.yaml

``` bash
---
- hosts: remote_hosts
  become: true
  roles:
    - nginx
```
***

* Role nginx main.yaml file

```bash
---
- name: nginx deploy for CentOS systems
  include: nginx_centos.yaml
  when: ansible_distribution == 'CentOS'

- name: nginx deploy for Ubuntu systems
  include: nginx_ubuntu.yaml
  when: ansible_distribution == 'Ubuntu'
nginx_ubuntu.yaml
- name: Create home directory
  file:
    path: /var/www/{{ item }}
    state: directory
  loop: "{{ virtual_hosts }}"

- name: Copy index file
  copy:
    src: index_{{ item }}.html
    dest: "{{ default_home }}/{{ item }}/index.html"
  loop: "{{ virtual_hosts }}"

- name: Modify host file
  template:
    src: "hosts.j2"
    dest: "/etc/hosts"
    backup: yes

- name: Create virtual host config
  template:
    src: "virt_host.conf.j2"
    dest: "/etc/nginx/conf.d/{{ item }}.conf"
  loop: "{{ virtual_hosts }}"
  notify:
    - restart nginx

- meta: flush_handlers

- name: connection check
  uri:
    url: "http://{{ item }}"
    return_content: yes
  register: webpage
  failed_when: "'Welcome to host' not in webpage.content"
  loop: "{{ virtual_hosts }}"
```

***

* Role nginx nginx_centos.yaml

``` bash
---
- name: Install EPEL Repo
  yum:
    name: epel-release
    state: present

- name: install the latest version of Nginx
  yum:
    name: "{{ package_to_install }}"
    state: latest

- name: Nginx. Enable and start servce
  service:
    name: "{{ package_to_install }}"
    state: started
    enabled: yes
```

***

* Role nginx nginx_ubuntu.yaml

```bash
---
- name: install the latest version of Nginx
  apt:
    name: "{{ package_to_install }}"
    state: latest

- name: Nginx. Enable and start servce
  service:
    name: "{{ package_to_install }}"
    state: started
    enabled: yes
```
